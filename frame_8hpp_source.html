466 // row-selector operator[]. One tricky bit is that rows(Ex) which it calls
467 // only requires is_expression<Ex> whereas here we require
468 // is_complex_expression<Ex>. This means that if we have a column (say
469 // column _1) that is bool-convertible, we can call
470 // fr.rows(_1);
471 // ...and that will select all rows where column _1 is 'true' but we CANNOT
472 // call
473 // fr[_1];
474 // ...because this is a column-selector call that returns a frame with a
475 // single column. Instead we either need to call rows() or call
476 // fr[_1 == true];
477 //
478 template<typename Ex>
479 std::enable_if_t<is_complex_expression<Ex>::value, frame<Ts...>>
480 operator[](Ex ex) const;
