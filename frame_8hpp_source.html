347 // row-selector operator[]. One tricky bit is that rows(Ex) which it calls
348 // only requires is_expression<Ex> whereas here we require
349 // is_complex_expression<Ex>. This means that if we have a column (say
350 // column _1) that is bool-convertible, we can call
351 // fr.rows(_1);
352 // ...and that will select all rows where column _1 is 'true' but we CANNOT
353 // call
354 // fr[_1];
355 // ...because this is a column-selector call that returns a frame with a
356 // single column. Instead we either need to call rows() or call
357 // fr[_1 == true];
358 //
359 template<typename Ex>
360 std::enable_if_t<is_complex_expression<Ex>::value, frame<Ts...>>
361 operator[](Ex ex) const;
